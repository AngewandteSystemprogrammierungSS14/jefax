33 static task_t* getNextTaskRR()
34 {
35  task_t *result;
36 
37  readyUpBlockingTasksRR();
38 
39  //ready list is empty
40  if(isEmpty(schedulerRR.readyList))
41  {
42  if(!hasRunningTask() || TASK_IS_BLOCKING(getRunningTask()))
43  result = NULL;
44  else {
45  getRunningTask()->state = RUNNING;
46  result = getRunningTask();
47  }
48  }
49  else {
50  // get next task with highest priority
51  result = getLast(schedulerRR.readyList);
52 
53  //next task would have lower prio, keep running task
54  if(hasRunningTask() && !TASK_IS_BLOCKING(getRunningTask()) && result->priority > getRunningTask()->priority) {
55  getRunningTask()->state = RUNNING;
56  result = getRunningTask();
57  } else {
58  popTaskBack(schedulerRR.readyList);
59 
60  if(hasRunningTask() && TASK_IS_RUNNING(getRunningTask()))
61  getRunningTask()->state = READY;
62  }
63  }
64 
65  //put runningTask in correct List
66  if(hasRunningTask() && TASK_IS_READY(getRunningTask()))
67  insertTaskRR(schedulerRR.readyList, getRunningTask());
68  else if(hasRunningTask() && TASK_IS_BLOCKING(getRunningTask()))
69  pushTaskBack(schedulerRR.blockingList, getRunningTask());
70 
71  return result;
72 }

74 static void readyUpBlockingTasksRR()
75 {
76  volatile int i;
77  // for all blocking task check if their state changed to non blocking
78  for(i = 0; i < schedulerRR.blockingList->count; ++i) {
79  if(!TASK_IS_BLOCKING(schedulerRR.blockingList->elements[i])) {
80  insertTaskRR(schedulerRR.readyList, schedulerRR.blockingList->elements[i]);
81  removeTask(schedulerRR.blockingList, i);
82  }
83  }
84 }

93 {

95

97  return 0;
98

100  while(result < p_tasks->count && p_tasks->elements[result]->priority > p_task->priority)

106 static void taskStateChangedRR(task_t* p_task)
107 {
108  //check if there is any running task, if true check if high prior task got ready
109  if(hasRunningTask() && TASK_IS_READY(p_task) && p_task->priority < getRunningTask()->priority)
110  getRunningTask()->state = READY;
111  // if running task is not in running state anymore switch task
112  if(!hasRunningTask() || !TASK_IS_RUNNING(getRunningTask()))
113  forceContextSwitch();
